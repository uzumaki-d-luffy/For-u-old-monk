function createBalloons(){

let area=document.getElementById("balloonArea");

let positions=[
{top:"20%",left:"20%"},
{top:"25%",left:"60%"},
{top:"35%",left:"80%"},
{top:"65%",left:"20%"},
{top:"70%",left:"55%"},
{top:"55%",left:"75%"}
];

for(let i=0;i<6;i++){

let balloon=document.createElement("div");

balloon.className="popBalloon";
balloon.innerHTML="🎈";

balloon.style.top=positions[i].top;
balloon.style.left=positions[i].left;

balloon.onclick=function(){
popBalloon(balloon,i);
};

area.appendChild(balloon);

}

}

function popBalloon(balloon,index){

if(popupOpen) return;

popupOpen = true;

balloon.classList.add("burst");

setTimeout(function(){
balloon.remove();
},300);

remaining--;

document.getElementById("remaining").innerText=remaining;

if(remaining === 0){
setTimeout(startFinalSequence,10000);
}
let popup=document.getElementById("surprisePopup");

document.getElementById("surpriseImg").src=surprises[index].img;
document.getElementById("surpriseText").innerText=surprises[index].text;

popup.style.display="flex";

setTimeout(function(){

popup.style.display="none";
popupOpen = false;

},10000);

}



function createHearts(){

let colors=["❤️","💖","💗","💓","💘"];

for(let i=0;i<25;i++){

let heart=document.createElement("div");

heart.className="heart";
heart.innerHTML=colors[Math.floor(Math.random()*colors.length)];

heart.style.left=Math.random()*100+"vw";
heart.style.fontSize=(20+Math.random()*30)+"px";

document.body.appendChild(heart);

setTimeout(()=>{
heart.remove();
},5000);

}

}

function confettiExplosion(){

for(let i=0;i<80;i++){

let conf=document.createElement("div");

conf.className="confetti";

let colors=["#ff4d6d","#ffcc00","#00e5ff","#7cff00","#ff6ec7"];

conf.style.background=colors[Math.floor(Math.random()*colors.length)];

let x=(Math.random()-0.5)*800+"px";
let y=(Math.random()-0.5)*600+"px";

conf.style.setProperty("--x",x);
conf.style.setProperty("--y",y);

document.body.appendChild(conf);

setTimeout(()=>{
conf.remove();
},2500);

}

}
